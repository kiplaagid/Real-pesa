let multiplier = 1.00; let isPlaying = false; let crashPoint = 0; let interval; const multiplierText = document.getElementById('multiplier'); const betBtn = document.getElementById('betBtn'); const cashOutBtn = document.getElementById('cashOutBtn'); function startRound() { multiplier = 1.00; isPlaying = true; // Random crash point between 1.1 and 10.0 crashPoint = (Math.random() * 9 + 1.1).toFixed(2); betBtn.disabled = true; cashOutBtn.disabled = false; interval = setInterval(() => { multiplier += 0.01; multiplierText.innerText = multiplier.toFixed(2) + "x"; if (multiplier >= crashPoint) { endRound("CRASHED!"); } }, 100); } function endRound(message) { clearInterval(interval); isPlaying = false; multiplierText.innerText = message + " at " + multiplier.toFixed(2) + "x"; betBtn.disabled = false; cashOutBtn.disabled = true; } betBtn.addEventListener('click', startRound); cashOutBtn.addEventListener('click', () => { if (isPlaying) { clearInterval(interval); isPlaying = false; multiplierText.innerText = "CASHOUT! " + multiplier.toFixed(2) + "x"; betBtn.disabled = false; cashOutBtn.disabled = true; } });
